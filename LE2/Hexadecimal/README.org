void opcao(node *PILHA, node *invertida, int op)
{
    node *tmp;
    switch (op)
    {
    case 0:
        libera(PILHA);
        break;

    case 1:
        libera(PILHA);
        libera(invertida);
        inicia(PILHA);
        inicia(invertida);
        break;

    case 2:
        exibe_int(PILHA);
        break;

    case 3:
        divide(PILHA, 44221);
        exibe_int(PILHA);
        break;

    case 4:
        tmp = pop(PILHA, invertida);
        if (tmp != NULL)
            printf("Retirado/Adicionado: %3d\n\n", tmp->num);
        break;

    default:
        printf("Comando invalido\n\n");
    }
}
